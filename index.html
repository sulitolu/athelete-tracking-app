  <div class="panel">

    <h2>Wellness Entry</h2>

    <label>Date <input type="date" id="wDate"></label>

    <label>Sleep Hours <input type="number" id="wSleep"></label>

    <label>Stress (1-5) <input type="number" id="wStress"></label>

    <label>Soreness (1-5) <input type="number" id="wSore"></label>

    <label>Mood (1-5) <input type="number" id="wMood"></label>

    <label>Weight (kg) <input type="number" id="wWeight"></label>

    <button onclick="saveWellness()">Save Wellness</button>

  </div>


  <div class="panel">

    <h2>GPS Session</h2>

    <label>Date <input type="date" id="gDate"></label>

    <label>Distance (m) <input type="number" id="gDist"></label>

    <label>Top Speed (m/s) <input type="number" id="gTS"></label>

    <label>RPE (1-10) <input type="number" id="gRPE"></label>

    <label>Duration (min) <input type="number" id="gDur"></label>

    <button onclick="saveGPS()">Save GPS</button>

  </div>


  <div class="panel">

    <h2>Gym Log</h2>

    <label>Date <input type="date" id="yDate"></label>

    <label>Lift <input type="text" id="yLift"></label>

    <label>Reps <input type="number" id="yReps"></label>

    <label>Weight (kg) <input type="number" id="yW"></label>

    <button onclick="saveGym()">Save Gym</button>

  </div>

    <h2>GPS Performance</h2>

    <canvas id="gpsChart"></canvas>

  </div>


  <script>

    const S = {

      wellness: JSON.parse(localStorage.getItem('wellness')||'[]'),

      gps: JSON.parse(localStorage.getItem('gps')||'[]'),

      gym: JSON.parse(localStorage.getItem('gym')||'[]')

    };

    const save = () => {

      localStorage.setItem('wellness', JSON.stringify(S.wellness));

      localStorage.setItem('gps', JSON.stringify(S.gps));

      localStorage.setItem('gym', JSON.stringify(S.gym));

      drawAll();

    };


    function saveWellness(){

      const r = {

        date: document.getElementById('wDate').value,

        sleep: +document.getElementById('wSleep').value,

        stress: +document.getElementById('wStress').value,

        sore: +document.getElementById('wSore').value,

        mood: +document.getElementById('wMood').value,

        weight: +document.getElementById('wWeight').value

      };

      S.wellness.push(r); save();

    }


    function saveGPS(){

      const r = {

        date: document.getElementById('gDate').value,

        dist: +document.getElementById('gDist').value,

        ts: +document.getElementById('gTS').value,

        rpe: +document.getElementById('gRPE').value,

        dur: +document.getElementById('gDur').value

      };

      r.srpe = r.rpe * r.dur;

      S.gps.push(r); save();

    }


    function saveGym(){

      const r = {

        date: document.getElementById('yDate').value,

        lift: document.getElementById('yLift').value,

        reps: +document.getElementById('yReps').value,

        w: +document.getElementById('yW').value

      };

      S.gym.push(r); save();

    }


    function drawAll(){ drawWeight(); drawGPS(); }

    function drawWeight(){

      const ctx=document.getElementById('weightChart');

      if(!ctx) return;

      const d=S.wellness; const labels=d.map(r=>r.date); const vals=d.map(r=>r.weight);

      if(window.wChart) window.wChart.destroy();

      window.wChart=new Chart(ctx,{type:'line',data:{labels,datasets:[{label:'Weight (kg)',data:vals}]}});

    }

    function drawGPS(){

      const ctx=document.getElementById('gpsChart');

      if(!ctx) return;

      const d=S.gps; const labels=d.map(r=>r.date); const dist=d.map(r=>r.dist); const ts=d.map(r=>r.ts);

      if(window.gChart) window.gChart.destroy();

      window.gChart=new Chart(ctx,{type:'line',data:{labels,datasets:[{label:'Distance (m)',data:dist},{label:'Top Speed (m/s)',data:ts}]}});

    }


    drawAll();

  </script>
